Problem 6: Copy the following code and save as dirtree.sh (do not copy "----------" at the beginning and end of the code)

----------

#!/bin/sh
# dirtree : print a hierarchy tree starting at specified directory
# Usage: dirtree


(cd ${1-.};pwd);
find ${1-.}  -print | sort -f | sed -e "s,^${1-.},," -e "/^$/d" -e "s,[^/]*/\([^/]*\)$,\`---\1," -e "s,[^/]*/,|   ,g"
